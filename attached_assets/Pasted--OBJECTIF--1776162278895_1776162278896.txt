3. HOOK : CALCULER TOTAUX
═══════════════════════════════════════════════════════════════════════════════

Créer `/src/hooks/useChargesSummary.ts` :

export const useChargesSummary = (charges: Charge[]) => {
  const totalPrevu = charges.reduce((sum, c) => sum + c.montantPrevu, 0);
  const totalReel = charges.reduce((sum, c) => sum + c.montantReel, 0);
  const totalRestant = totalPrevu - totalReel;
  
  return {
    totalPrevu: parseFloat(totalPrevu.toFixed(2)),
    totalReel: parseFloat(totalReel.toFixed(2)),
    totalRestant: parseFloat(totalRestant.toFixed(2)),
  };
};

4. COMPOSANT : RECAP (HAUT DE LA PAGE CHARGES)

interface ChargesRecapProps {
  totalPrevu: number;
  totalReel: number;
  totalRestant: number;
}

export const ChargesRecap: React.FC<ChargesRecapProps> = ({
  totalPrevu,
  totalReel,
  totalRestant,
}) => {
  const percentUsed = totalPrevu > 0 ? (totalReel / totalPrevu) * 100 : 0;
  
  return (
    <div className="bg-gradient-to-r from-blue-50 to-indigo-50 p-6 rounded-lg mb-6 border border-blue-200">
      <h2 className="text-lg font-bold text-gray-800 mb-4">📊 Synthèse Charges</h2>
      
      <div className="grid grid-cols-3 gap-4">
        <div>
          <p className="text-sm text-gray-600">Prévu</p>
          <p className="text-2xl font-bold text-blue-600">{totalPrevu.toFixed(2)}€</p>
        </div>
        
        <div>
          <p className="text-sm text-gray-600">Réel</p>
          <p className={`text-2xl font-bold ${totalReel > totalPrevu ? 'text-red-600' : 'text-green-600'}`}>
            {totalReel.toFixed(2)}€
          </p>
          <p className="text-xs text-gray-500 mt-1">{percentUsed.toFixed(0)}% utilisé</p>
        </div>
        
        <div>
          <p className="text-sm text-gray-600">Restant</p>
          <p className={`text-2xl font-bold ${totalRestant >= 0 ? 'text-green-600' : 'text-red-600'}`}>
            {totalRestant.toFixed(2)}€
          </p>
        </div>
      </div>
      
      {totalReel > totalPrevu && (
        <div className="mt-4 p-3 bg-red-100 text-red-700 rounded text-sm">
          ⚠️ Budget dépassé de {Math.abs(totalRestant).toFixed(2)}€
        </div>
      )}
    </div>
  );
};

5. PAGE CHARGES : INTÉGRATION
═══════════════════════════════════════════════════════════════════════════════

Modifier le composant Charges.tsx :

import { chargesData } from "@/data/chargesData";
import { useChargesSummary } from "@/hooks/useChargesSummary";
import { ChargesRecap } from "@/components/ChargesRecap";

const Charges = () => {
  const [charges, setCharges] = useState<Charge[]>([]);
  
  // Récupérer les charges depuis Firebase au mount
  useEffect(() => {
    const fetchCharges = async () => {
      const user = auth.currentUser;
      if (!user) return;
      
      const chargesRef = collection(db, "users", user.uid, "charges");
      const snap = await getDocs(chargesRef);
      
      if (snap.empty) {
        // PREMIÈRE CONNEXION : injecter les 34 catégories
        const promises = chargesData.map((charge) =>
          addDoc(chargesRef, {
            ...charge,
            montantReel: 0,
            montantRestant: charge.montantPrevu,
            createdAt: new Date(),
            updatedAt: new Date(),
          })
        );
        await Promise.all(promises);
        
        const newSnap = await getDocs(chargesRef);
        setCharges(newSnap.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      } else {
        setCharges(snap.docs.map(doc => ({ id: doc.id, ...doc.data() })));
      }
    };
    
    fetchCharges();
  }, []);
  
  // Calculer les totaux
  const { totalPrevu, totalReel, totalRestant } = useChargesSummary(charges);
  
  // Trier alphabétiquement
  const sortedCharges = [...charges].sort((a, b) => a.name.localeCompare(b.name));
  
  // Modifier une charge
  const handleUpdateCharge = async (chargeId: string, montantReel: number) => {
    const user = auth.currentUser;
    if (!user) return;
    
    const chargeRef = doc(db, "users", user.uid, "charges", chargeId);
    const montantRestant = sortedCharges.find(c => c.id === chargeId)?.montantPrevu - montantReel;
    
    await updateDoc(chargeRef, {
      montantReel,
      montantRestant,
      updatedAt: new Date(),
    });
    
    // Mettre à jour l'état local
    setCharges(charges.map(c =>
      c.id === chargeId
        ? { ...c, montantReel, montantRestant: montantRestant || 0 }
        : c
    ));
    
    showToast("✓ Charge mise à jour", "success");
  };
  
  return (
    <div className="p-4">
      <ChargesRecap
        totalPrevu={totalPrevu}
        totalReel={totalReel}
        totalRestant={totalRestant}
      />
      
      <div className="bg-white rounded-lg border border-gray-200">
        <table className="w-full text-sm">
          <thead>
            <tr className="border-b bg-gray-50">
              <th className="text-left p-3 font-semibold">Catégorie</th>
              <th className="text-right p-3 font-semibold">Prévu</th>
              <th className="text-right p-3 font-semibold">Réel</th>
              <th className="text-right p-3 font-semibold">Restant</th>
            </tr>
          </thead>
          <tbody>
            {sortedCharges.map((charge) => (
              <tr key={charge.id} className="border-b hover:bg-gray-50">
                <td className="p-3">{charge.name}</td>
                <td className="text-right p-3 text-blue-600 font-semibold">
                  {charge.montantPrevu.toFixed(2)}€
                </td>
                <td className="text-right p-3">
                  <input
                    type="number"
                    value={charge.montantReel}
                    onChange={(e) =>
                      handleUpdateCharge(charge.id, parseFloat(e.target.value) || 0)
                    }
                    className="w-24 px-2 py-1 border rounded text-right"
                    step="0.01"
                  />
                </td>
                <td
                  className={`text-right p-3 font-semibold ${
                    charge.montantRestant >= 0 ? 'text-green-600' : 'text-red-600'
                  }`}
                >
                  {charge.montantRestant.toFixed(2)}€
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
};

export default Charges;
